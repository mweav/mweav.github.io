Categories with Families...in Agda!

Posted on February 2, 2016 by Matthew Weaver

Here I have a neat definition of Categories with Families I implemented in Agda. A Category with Families (CwF) is a categorical model of dependent type theory.

module CwF where

open import Level
open import Categories.Category
open import Categories.Functor renaming (_≡_ to _==_; _∘_ to comp)
open import Categories.Fam 
open import Categories.Presheaf
open import Categories.Support.PropositionalEquality

import Categories.Object.Terminal as Terminal

data _≅_ {i} {A : Set i} (a : A) : {B : Set i} → B → Set i where
  refl : a ≅ a


coe : ∀ {i}
      {X : Set i}
      {A B : X}
      {F : X → Set i}
      (p : A ≅ B)
      → -------------
      F A ≅ F B
      
coe refl = refl


tr : ∀ {i}
       {A B : Set i}
       (p : A ≅ B)
       → ---------
       A → B
            
tr refl a = a


record CwF {o ℓ e a b : Level}
       (C : Category o ℓ e)
       (F : Presheaf C (Fam a b))
       : Set (suc (o ⊔ ℓ ⊔ e ⊔ a ⊔ b)) where

  open Category C
  open Terminal C
  open Functor F
  open Fam renaming (_≡_ to _==_; _∘_ to comp)

  module C = Category C
  module F = Functor F

  Ctx = C.Obj


  Subst = C._⇒_


  Id : {Γ : Ctx} → Subst Γ Γ
  Id = C.id


  Ty : Ctx → Set _
  Ty Γ = U (F.F₀ Γ)


  Ter : (Γ : Ctx) → Ty Γ → Set _
  Ter Γ A = (T (F.F₀ Γ)) A


  TApp : {Δ Γ : Ctx}
         (σ   : Subst Δ Γ)
         (A   : Ty Γ)
         → ---------------
         Ty Δ
         
  TApp σ A = (Hom.f (F.F₁ σ)) A


  tApp : {Δ Γ : Ctx}
         (σ   : Subst Δ Γ)
         {A   : Ty Γ}
         (a   : Ter Γ A)
         → ---------------
         Ter Δ (TApp σ A)
         
  tApp σ {A} a = (Hom.φ (F.F₁ σ)) A a


  field
    TerIdEq   : {Γ : Ctx}
                {A     : Ty Γ}
                → --------------------------------------------------------
                Ter Γ A ≅ Ter Γ (TApp Id A)

    TerAppEq  : {Δ Γ Θ : Ctx}
                {σ     : Subst Δ Γ}               
                {τ     : Subst Θ Δ}
                {A     : Ty Γ}
                → --------------------------------------------------------
                Ter Θ (TApp τ (TApp σ A)) ≅ Ter Θ (TApp (σ ∘ τ) A)

    TAppEq    : {Δ Γ Θ : Ctx}
                {σ     : Subst Δ Γ}               
                {τ     : Subst Θ Δ}
                {A     : Ty Γ}
                → --------------------------------------------------------
                TApp τ (TApp σ A) ≅ TApp (σ ∘ τ) A

    tAppEq    : {Δ Γ Θ : Ctx}
                {σ     : Subst Δ Γ}               
                {τ     : Subst Θ Δ}
                {A     : Ty Γ}
                {a     : Ter Γ A}
                → --------------------------------------------------------
                tApp τ (tApp σ a) ≅ tApp (σ ∘ τ) a

    []        : Terminal

    ctxExt    : (Γ : Ctx)
                (A : Ty Γ)
                → --------------------------------------------------------
                Ctx
             
    p         : {Γ : Ctx}
                {A : Ty Γ}
                → --------------------------------------------------------
                Subst (ctxExt Γ A) Γ

    q         : {Γ : Ctx}
                {A : Ty Γ}
                → --------------------------------------------------------
                Ter (ctxExt Γ A) (TApp (p {Γ} {A}) A)

    uSubst    : {Δ Γ : Ctx}
                (σ   : Subst Δ Γ)
                {A   : Ty Γ}
                (a   : Ter Δ (TApp σ A))
                → --------------------------------------------------------
                Subst Δ (ctxExt Γ A)

    uEq₁      : {Δ Γ : Ctx}
                (σ   : Subst Δ Γ)
                {A   : Ty Γ}
                (a   : Ter Δ (TApp σ A))
                → --------------------------------------------------------
                p ∘ (uSubst σ a) ≡ σ

    uEq₂      : {Γ : Ctx}
                {A : Ty Γ}
                → --------------------------------------------------------
                uSubst (p {Γ} {A}) (q {Γ} {A}) ≡ Id
                
    uEq₃      : {Δ Γ Θ : Ctx}
                (σ     : Subst Δ Γ)
                {A     : Ty Γ}
                (a     : Ter Δ (TApp σ A))
                (τ     : Subst Θ Δ)
                → --------------------------------------------------------
                (uSubst σ a) ∘ τ ≡ uSubst (σ ∘ τ) (tr TerAppEq (tApp τ a))

    uEq₄      : {Δ Γ : Ctx}
                (σ   : Subst Δ Γ)
                {A   : Ty Γ}
                (a   : Ter Δ (TApp σ A))
                → --------------------------------------------------------
                tApp (uSubst σ a) q ≅ a
